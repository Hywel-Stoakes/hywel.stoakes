# Preamble
library(plot3D)
# Reduce the resolution
 Volcano <- volcano[seq(1, nrow(volcano), by = 3),
                     seq(1, ncol(volcano), by = 3)]
image2D(Volcano, lighting = TRUE, rasterImage = TRUE,
     contour = list(col = "white", labcex = 0.8, lwd = 3, alpha = 0.5))
